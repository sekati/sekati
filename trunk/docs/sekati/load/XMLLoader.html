XMLLoader handles the loading of XML documents.


Example 
  var loader : XMLLoader = new XMLLoader( "my.xml" );
  loader.addEventListener ( XMLEvent.ON_READY, xmlHandler );
  var xmlHandler : Function = function(e:XMLEvent):void {
    trace( "w00t:\n\n" + e.data );
  };
  loader.request( );
